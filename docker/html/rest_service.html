The IRIS RESTful service enables the access of the translation system with other applications. For any questions, contact Mihael Arcan, PhD Student at Unit for Natural Language Processing (UNLP), Insight @ NUI Galway.


Example: Translating English text into Irish within java/python/perl code:

   "text":"Ireland will use no fossil fuels by the end of the century, according to a new Government energy policy paper.",

}

    JAVA EXAMPLE:

    import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.net.MalformedURLException;
import java.net.URL;
import java.net.URLConnection;
import java.net.URLEncoder;

public class RestApiExample {
	static String endpoint = "http://server1.nlp.insight-centre.org/cgi-bin/iris_restAPI.cgi?json=";
	@SuppressWarnings("deprecation")
	public static String callApi(){
		String extractText = "{\"text\":\"Ireland will use no fossil fuels by the end of the century, according to a new Government energy policy paper.\",\"translation_direction\":\"en_ga\",\"nbest\":10}";
		extractText = URLEncoder.encode(extractText).replace("+", "%20");
		String query = endpoint + extractText;
		StringBuffer buffer = new StringBuffer();
		try {
			URL url = new URL(query);
			URLConnection connection = url.openConnection();
			BufferedReader reader = new BufferedReader(new InputStreamReader(connection.getInputStream()));			
			String inputLine = null;
			while ((inputLine = reader.readLine()) != null){ 
				buffer.append(inputLine);
			}
			reader.close();			
		} catch (MalformedURLException e) {
			e.printStackTrace();
		} catch (IOException e) {
			e.printStackTrace();
		}
		return buffer.toString();
	}
	public static void main(String[] args) {
		String outJson = callApi();		
		System.out.println(outJson);
	}
}
